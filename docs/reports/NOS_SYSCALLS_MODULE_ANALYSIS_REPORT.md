    pub fn resolve_dependencies(&self, service_name: &str) -> Result<Vec<(String, Arc<dyn Any>)>> {
        let mut resolved = HashSet::new();
        let mut visiting = HashSet::new();

        self.resolve_service_dependencies(service_name, &mut resolved, &mut visiting)?;

        let dependencies = resolved.into_iter()
            .filter(|name| name != service_name)
            .map(|name| {
                let service = self.services.get(&name).unwrap();
                (name, Arc::clone(&service.service))
            })
            .collect();

        Ok(dependencies)
    }

    fn resolve_service_dependencies(
        &self,
        service_name: &str,
        resolved: &mut HashSet<String>,
        visiting: &mut HashSet<String>
    ) -> Result<()> {
        if resolved.contains(service_name) {
            return Ok(());
        }

        if visiting.contains(service_name) {
            return Err(Error::CircularDependency(service_name.to_string()));
        }

        visiting.insert(service_name.to_string());

        let service = self.services.get(service_name)
            .ok_or_else(|| Error::ServiceNotFound(service_name.to_string()))?;

        for dep in &service.metadata.dependencies {
            self.resolve_service_dependencies(dep, resolved, visiting)?;
        }

        visiting.remove(service_name);
        resolved.insert(service_name.to_string());
        Ok(())
    }
}
